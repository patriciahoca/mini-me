const canvas = document.getElementById("game"); const ctx = canvas.getContext("2d"); let fundoImg = null; let bonecaImg = null; const imgs = []; let maxZ = 0; let drag = null; function loadImg(src) { return new Promise((res, rej) => { const img = new Image(); const cx = c.getContext("2d"); cx.drawImage(img, 0, 0); const d = cx.getImageData(0, 0, c.width, c.height); return { data: d.data, w: c.width, h: c.height }; } catch { return null; } } function getAlpha(cache, lx, ly, cw, ch) { if (!cache) return -1; const ix = Math.round((lx / cw) * cache.w); const iy = Math.round((ly / ch) * cache.h); if (ix < 0 || ix >= cache.w || iy < 0 || iy >= cache.h) return 0; return cache.data[(iy * cache.w + ix) * 4 + 3]; const sc = canvas.width / r.width; const src = e.touches ? e.touches[0] : e; return { x: (src.clientX - r.left) * sc, y: (src.clientY - r.top) * sc, clientX: src.clientX, clientY: src.clientY, }; } function findItem(px, py) { const w = canvas.width, h = canvas.height; window.addEventListener("touchend", () => { drag = null; }); canvas.addEventListener("pointerdown", e => { if (e.pointerType !== "touch") onStart(e, false); }); window.addEventListener("pointermove", e => { if (e.pointerType !== "touch") onMove(e, false);
