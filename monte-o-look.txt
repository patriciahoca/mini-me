  <style>
    * { margin: 0; padding: 0; box-sizing: border-box; }
    html, body {
      width: 100%; height: 100%;
      background: #c9a8f0;
      display: flex;
      align-items: center;
      justify-content: center;
      overflow: hidden;
    }
    canvas {
      display: block;
      cursor: grab;
      touch-action: none;
      user-select: none;
    }
    canvas:active { cursor: grabbing; }
  </style>

<body>
<canvas id="game"></canvas>
<script>
  const FUNDO  = "assets/fundo2.png";
  const BONECA = "assets/Bonequinha.png";
  const ITEMS  = [
    "assets/roupa2.png",
    "assets/roupa.png",
    "assets/roupa%20leia.png",
    "assets/saia.png",
    "assets/allstar.png",
    "assets/sapato.png",
    "assets/cabelo.png",
    "assets/cabelo%20leia.png",
    "assets/oculos%20verdes.png",
    "assets/oculos%20oncinha.png",
    "assets/tv.png",
    "assets/livro.png",
    "assets/bola.png",
    "assets/bola%20tenis.png",
    "assets/bolsa.png",
    "assets/cafe.png",
    "assets/cha.png",
    "assets/cervejinha.png",
    "assets/natacao.png",
    "assets/partitura.png",
    "assets/quadro.png",
  ];

  const canvas = document.getElementById("game");
  const ctx    = canvas.getContext("2d");

  let fundoImg  = null;
  let bonecaImg = null;
  const imgs    = [];
  let maxZ      = 0;
  let drag      = null;

  function loadImg(src) {
    return new Promise(res => {
      const img = new Image();
      img.crossOrigin = "anonymous";
      img.onload = () => res(img);
      img.src = src;
    });
  }

  function buildCache(img) {
    try {
      const c = document.createElement("canvas");
      c.width  = img.naturalWidth;
      c.height = img.naturalHeight;
      const cx = c.getContext("2d");
      cx.drawImage(img, 0, 0);
      const d = cx.getImageData(0, 0, c.width, c.height);
      return { data: d.data, w: c.width, h: c.height };
    } catch { return null; }
  }

  function getAlpha(cache, lx, ly, cw, ch) {
    if (!cache) return 255;
    const ix = Math.round((lx / cw) * cache.w);
    const iy = Math.round((ly / ch) * cache.h);
    if (ix < 0 || ix >= cache.w || iy < 0 || iy >= cache.h) return 0;
    return cache.data[(iy * cache.w + ix) * 4 + 3];
  }

  function draw() {
    const w = canvas.width, h = canvas.height;
    ctx.clearRect(0, 0, w, h);
    if (fundoImg)  ctx.drawImage(fundoImg,  0, 0, w, h);
    if (bonecaImg) ctx.drawImage(bonecaImg, 0, 0, w, h);
    [...imgs]
      .sort((a, b) => a.z - b.z)
      .forEach(item => ctx.drawImage(item.img, item.x, item.y, w, h));
  }

  function resize() {
    if (!fundoImg) return;
    const aspect = fundoImg.naturalWidth / fundoImg.naturalHeight;
    const vw = window.innerWidth, vh = window.innerHeight;
    const w  = Math.min(vw, vh * aspect);
    const h  = w / aspect;
    canvas.style.width  = w + "px";
    canvas.style.height = h + "px";
    canvas.width  = Math.round(w);
    canvas.height = Math.round(h);
    draw();
  }

  canvas.addEventListener("pointerdown", e => {
    e.preventDefault();
    const r  = canvas.getBoundingClientRect();
    const sx = canvas.width  / r.width;
    const sy = canvas.height / r.height;
    const px = (e.clientX - r.left) * sx;
    const py = (e.clientY - r.top)  * sy;
    const w  = canvas.width, h = canvas.height;

    const sorted = imgs
      .map((item, idx) => ({ item, idx }))
      .sort((a, b) => b.item.z - a.item.z);

    for (const { item, idx } of sorted) {
      if (getAlpha(item.cache, px - item.x, py - item.y, w, h) > 30) {
        item.z = ++maxZ;
        drag = { idx, sx: e.clientX, sy: e.clientY, ox: item.x, oy: item.y };
        canvas.style.cursor = "grabbing";
        draw();
        break;
      }
    }
  });

  window.addEventListener("pointermove", e => {
    if (!drag) return;
    const r  = canvas.getBoundingClientRect();
    const sc = canvas.width / r.width;
    const item = imgs[drag.idx];
    item.x = drag.ox + (e.clientX - drag.sx) * sc;
    item.y = drag.oy + (e.clientY - drag.sy) * sc;
    draw();
  });

  window.addEventListener("pointerup", () => {
    drag = null;
    canvas.style.cursor = "grab";
  });

  window.addEventListener("resize", resize);

  Promise.all([
    loadImg(FUNDO).then(img  => { fundoImg  = img; }),
    loadImg(BONECA).then(img => { bonecaImg = img; }),
    ...ITEMS.map((src, i) =>
      loadImg(src).then(img => {
        imgs[i] = { img, cache: buildCache(img), x: 0, y: 0, z: i };
      })
    ),
  ]).then(() => {
    maxZ = ITEMS.length;
    resize();
  });
</script>
</body>
</html>
